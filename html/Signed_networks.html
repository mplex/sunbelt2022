Structural Balance

Balance semiring structures in a signed triad

# create a triadic signed structure with three valences (sort vertices labels)
sgn <- transf(list(p=c("1, 3", "3, 1", "2, 3"), n=c("2, 1", "2, 3")), type="toarray", sort=TRUE)
# re-define node / edge / graph characteristics
scps <- list(cex=10, vcol=1, ecol=1, lwd=8, bwd=.75, swp=TRUE, fsize=45, rot=-30)
# signed structure 'sgn' 
multigraph(sgn, signed=TRUE, scope=scps)
